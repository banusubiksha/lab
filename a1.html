    Area Calculator

    Calculate Area



    Result

    Index.js // index.js const express = require('express'); const app = express(); const PORT = process.env.PORT || 3000; // Middleware app.use(express.json()); app.use(express.urlencoded({ extended: true })); // Serve HTML file app.get('/', (req, res) => { res.sendFile(__dirname + '/index.html'); }); // Route for calculating area app.post('/calculate-area', (req, res) => { const { shape, dimensions } = req.body; if (!shape || !dimensions || !Array.isArray(dimensions)) { return res.status(400).json({ error: 'Invalid input' }); } let area; switch (shape.toLowerCase()) { case 'rectangle': if (dimensions.length !== 2) { return res.status(400).json({ error: 'Invalid dimensions for rectangle' }); } const [length, width] = dimensions; area = length * width; break; case 'triangle': if (dimensions.length !== 2) { return res.status(400).json({ error: 'Invalid dimensions for triangle' }); } const [base, height] = dimensions; area = 0.5 * base * height; break; case 'circle': if (dimensions.length !== 1) { return res.status(400).json({ error: 'Invalid dimensions for circle' }); } const [radius] = dimensions; area = Math.PI * radius * radius; break; default: return res.status(400).json({ error: 'Unsupported shape' }); } res.json({ shape, area }); }); // Start the server app.listen(PORT, () => { console.log(`Server is running on port ${PORT}`); }); 3.
